# Refresher of Linux Commands 

- It's basically just learning the linux commands again 
- Like here [[General commands]]
- so starting over 

- pwd: find working directory 
- ls : basic listing 
- ls -l : detailed information 
- ls -a : show hidden files 
- ls -la : detailed info with hidden files 
- cd : change directory 
- cd -- : up one level 
- cd ~ : home 
- clear: clear 
- whoami: displays the current user username 
- which: shows where a command is located in the system
	- which ls : /usr/bin/ls
- mkdir: make directory 
- mkdir -p something/somethinginside: making nested directories 
- mkdir foder1 folder2: creating multiple directories
- rmdir folder: remove folders 
- touch file.txt : create file
- touch file1.txt file2.txt : create multiple files 
- cp file.txt anotherfile.txt : copy files 
- cp -r folder1 folder2: copy folders
- cp \*.txt backup/ : copy all txt files 
- mv file.txt Documents/ : move file 
- mv file.txt newname.txt: rename file 
- mv folder1 folder2: rename folder 
- rm file.txt: remove file 
- rm -r folder: remove folder and contents 
- rm -f file.txt: force remove - no confirmation 
- rm \*txt: Delete all .txt files 
- find . -name "filname" : find file by name 
- find /home -name "\*.txt" : find all .txt files 
- find -type d -name temp: find all folders wtih the name temp
	- -type mentions what type of file we are searching for 
	- d means directory 
- find -type d -name "temp*": find all folders wtih the name starting with temp
- find . -size +100MIf you delete or move the original file/directory, the symbolic link will become "broken" or "dangling" because it's still pointing to a non-existent path.
- locate filname.txt : faster find 
- locate "\*pdf"
- ln sourcefile link_name : creates hard link to file.txt - [[ln command]]
- ln -s file.txt softlink.txt: creates symbolic link (shortcut)
- ln -s "/pathtofolder" linkfolder: link to folder 
- cat file.txt : display file content all at once 
- cat file1.txt file2.txt : showing multiple files 
- less file1.txt : show file content one screen at a time 
- head file.txt: show first 10 lines of a file 
- head -n 5 file.txt: show first 5 lines of a file 
- tail file.txt: show last 10 lines of the file 
- tail -n 5 file.txt: show last 5 lines of the file 
- tail -f log.txt: follow file changes (great for logging)
- grep "hello" file.txt: find hello in file
- grep -i "HELLO" file.txt: case-insensitive search 
- grep -r "error" /var/log: search recursively
- grep -v "pattern" file.txt: show lines NOT matching the pattern 
- grep -C "pattern" file.txt: show two lines before and after match 
- grep -rli "pattern" . : find files containing that pattern 
- echo "hi" : print 
- echo "hi" > file.txt : save the text into the file (overwrites)
- echo "hi" >> file.txt : concatenate the text to the end of the file 
- wc file.txt : shows lines, words, characters 
- wc -l file.txt : shows lines
- wc -w file.txt : shows words 
- wc -c file.txt : shows characters
- sort file.txt : sorts alphabetically 
- sort -n file.txt : sorts numerically 
- sort -r file.txt : sorts in reverse order 
- sort -u file.txt : sorts and removes duplicates 
- tar -cvf archive.tar files/ :Create archive
- tar -xvf archive.tar :Extract archive
- tar -tvf archive.tar :List contents
- tar -czvf archive.tar.gz files/ :Create compressed archive
- tar -xzvf archive.tar.gz  :Extract compressed archive
- gzip file.txt : compress file
- gzip -d file.txt.gz : decompress file
- zip -r archive.zip folder/
- zip -d archive.zip file.txt 
- unzip archive.zip 
- unrzip -l archive.zip : list contents in the zip 
- unzip -d /path/to/extract archive.zip :extract to specific directory 
- ps : display your processes running 
- ps aux : all processes

- ps -e -o pcpu,cpu,nice,state,cputime,args --sort pcpu | head -n 10  # Top CPU processes
- top - overall system monitor 
- top -u username - monitors specific user 
- kill 1234 : kill process by pid
- kill -9 1234 : force kill 
- killall firefox : kill all firefox processes

- df -i : includes inode usage
